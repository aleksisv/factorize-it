6    * Class whose instantiations are polynomials, i.e of the form P(x) = 5x^3 + x^2

17      public Polynomial() {
18          this.coefficients = new long[1];

23      /**
24       * Default constructor.
25       *
26       * @param coeffs The coefficients of the polynomial.
27       */
28      public Polynomial(long... coeffs) {
29          this.coefficients = new long[coeffs.length];
30          for (int i = 0; i < coeffs.length; i++) {
31              coefficients[coefficients.length - 1 - i] = coeffs[i];
32          }
33  
34          this.degree = determineDegree();
35          this.sanitizePolynomial();
36      }

47      /**
48       * Reverses an array.
49       *
50       * @param arr Array to be reversed.
51       * @return Reversed array.
52       */
53      private long[] reverse(long[] arr) {
54          long[] reversed = new long[arr.length];
55          for (int i = 0; i < arr.length; i++) {
56              reversed[i] = arr[arr.length - i - 1];
57          }
58          return reversed;
59      }

61      /**
62       * Adds two polynomials together.
63       *
64       * @param p Polynomial to be added to this polynomial.
65       * @return The sum of two polynomials.
66       */
67      public Polynomial addPolynomial(Polynomial p) {
68  
69          int max = Math.max(this.getCoefficients().length, p.getCoefficients().length);
70          long initialArray[] = new long[max];
71  
72          for (int i = 0; i < initialArray.length; i++) {
73              if (this.coefficients.length <= i) {
74                  initialArray[i] = p.coefficients[i];
75              } else if (p.coefficients.length <= i) {
76                  initialArray[i] = this.coefficients[i];
77              } else {
78                  initialArray[i] = p.coefficients[i] + this.coefficients[i];
79              }
80          }
81  
82          long[] initialReversed = reverse(initialArray);
83  
84          Polynomial newPoly = new Polynomial(initialReversed);
85          newPoly.sanitizePolynomial();
86          return newPoly;
87      }

90       * Subtracts one polynomial from the other.
91       *
92       * @param poly Polynomial to be added to this polynomial.
93       * @return The sum of two polynomials.

95      public Polynomial subtract(Polynomial poly) {
96          return this.addPolynomial(poly.scalePolynomial(-1));

101      *
102      * @param scale The scaling factor.
103      * @return The scaled polynomial.
104      */
105     public Polynomial scalePolynomial(long scale) {
106         long initialArray[] = new long[this.coefficients.length];
107         for (int i = 0; i < initialArray.length; i++) {
108             initialArray[initialArray.length - i - 1] = scale * this.coefficients[i];
109         }
110 
111         Polynomial newPoly = new Polynomial(initialArray);
112         newPoly.sanitizePolynomial();
113         return newPoly;

116     /**
117      * Determines the degree of a polynomial.
118      *
119      * @return The degree of the polynomial in question.
120      */
121     private int determineDegree() {
122         int degree = 0;
123         for (int i = 0; i < this.coefficients.length; i++) {
124             if (this.coefficients[i] != 0) {
125                 degree = i;
126             }
127         }
128         return degree;
129     }

131     /**
132      * Sanitizes this polynomial, i.e removes the leading zeroes.
133      */
134     public void sanitizePolynomial() {
135 
136         while (this.coefficients[(int) this.coefficients.length - 1] == 0) {
137 
138             if (this.coefficients.length == 1) {
139                 return;
140             }
141             long[] newArray = new long[this.coefficients.length - 1];
142             for (int i = 0; i < newArray.length; i++) {
143                 newArray[i] = this.coefficients[i];
144             }
145             this.setCoefficients(newArray);
146         }
147 
148     }

150     /**
151      * Multiplies two polynomials.
152      *
153      * @param p the polynomial to be multiplied by this polynomial.
154      * @return The product of the two polynomials.
155      */
156     public Polynomial multiply(Polynomial p) {
157         long initialArray[] = new long[p.coefficients.length + this.coefficients.length];
158 
159         long[] thisReversed = reverse(this.coefficients);
160         long[] thatReversed = reverse(p.coefficients);
161 
162         for (int i = 0; i < thisReversed.length; i++) {
163             for (int j = 0; j < thatReversed.length; j++) {
164                 initialArray[i + j] += (this.coefficients[i] * p.coefficients[j]);
165 
166             }
167         }
168 
169         long[] initialReversed = reverse(initialArray);
170         Polynomial n = new Polynomial(initialReversed);
171         n.sanitizePolynomial();
172         return n;
173     }

175     /**
176      * Divides the current polynomial by some other polynomial.
177      *
178      * @param p The diviing polynomial.
179      * @return The result of the division.
180      */
181     public Polynomial longDivision(Polynomial p) {
182 
183         if (p.degree == 0 && p.coefficients[0] == 0) {
184             throw new RuntimeException("You cannot divide by the polynomial P(x)=0");
185         }
186 
187         if (this.degree < p.getDegree()) {
188             return new Polynomial();
189         }
190 
191         long numeratorVal = this.coefficients[this.degree];
192         long denumVal = p.coefficients[p.degree];
193 
194         long coef = numeratorVal / denumVal;
195         long exponent = this.degree - p.getDegree();
196 
197         Polynomial c = new Polynomial(1, 0);
198         Polynomial e = c.exponentiation(exponent);
199         Polynomial f = e.scalePolynomial(coef);
200 
201         Polynomial bc = p.multiply(f);
202         Polynomial mbc = bc.scalePolynomial(-1);
203         Polynomial ambc = this.addPolynomial(mbc);
204         ambc.sanitizePolynomial();
205         Polynomial ambcdb = ambc.longDivision(p);
206         Polynomial res = f.addPolynomial(ambcdb);
207 //        res.sanitizePolynomial();
208 
209         return res;
210     }

213      * Shifts polynomial to right by some number of steps.
214      *
215      * @param steps The number of steps.
216      * @return The shifted polynomial.
217      */
218     public Polynomial shiftRight(int steps) {
219         long[] initialArray = new long[this.coefficients.length + steps];
220 
221         for (int i = 0; i < this.coefficients.length; i++) {
222             initialArray[i] = this.coefficients[this.coefficients.length - 1 - i];
223         }
224 
225         Polynomial p = new Polynomial(initialArray);
226         p.sanitizePolynomial();
227         return p;

231     /**
232      * Uses Polynomial.multiply to calculate powers of the given exponential.
233      *
234      * @param exponent The exponent used to calculate the power of the
235      * polynomial.
236      * @return The resulting polynomial.
237      */
238     public Polynomial exponentiation(long exponent) {
239         Polynomial newpoly = new Polynomial(1);
240         while (exponent > 0) {
241             newpoly = newpoly.multiply(this);
242             exponent--;
243         }
244         return newpoly;
245     }

248     /**
249      * Uses Polynomial.multiply() and Polynomial.mod() to calculate the exponential
250      * of the polynomial when we take into account some modulus.
251      *
252      * @param exponent The exponent used to calculate the power of the
253      * polynomial.
254      * @param modulo Modulo, modulo which we are calculating.
255      * @return The resulting polynomial.
256      */
257     public Polynomial modularExponentiation(long exponent, long modulo) {
258         Polynomial copyPoly = this.copyThisPoly();
259         Polynomial resultPoly = this.copyThisPoly();
260         while(exponent>1) {
261             resultPoly = resultPoly.multiply(copyPoly);
262             resultPoly = resultPoly.mod(modulo);
263             exponent--;
264         }
265         resultPoly.sanitizePolynomial();
266         return resultPoly;
267     }

269     /**
270      * Uses calculates the congruence of polynomial modulo some other polynomial.
271      *
272      * @param polmod The modulo. This is of the form x^n - a.
273      * @return The resulting polynomial.
274      */
275     public Polynomial mod(Polynomial polmod) {
276         Polynomial remainder = new Polynomial(this.coefficients);
277 
278         while (remainder.degree >= polmod.degree) {
279             long difference = remainder.degree - polmod.degree;
280             Polynomial substitution = new Polynomial(polmod.coefficients)
281                     .multiply(new Polynomial(1,0)).exponentiation(difference);
282 
283             remainder = remainder.subtract(substitution);
284         }
285 
286         return remainder;
287     }

296     public Polynomial mod(long n) {
297         for (int i = 0; i < this.coefficients.length; i++) {
298             this.coefficients[i] = this.coefficients[i] % n;
299         }
300         this.sanitizePolynomial();
301         return this;

304     /**
305      * Checks if two polynomials are the same.
306      *
307      * @param p The polynomial against which we are comparing this one.
308      * @return Truth value. If true, the polynomials are equal. If false, they are not.
309      */
310     public boolean polyEquals(Polynomial p) {
311         if(this.coefficients.length != p.getCoefficients().length) {
312             return false;
313         }
314         
315         for (int i = 0; i < this.coefficients.length; i++) {
316             if(this.coefficients[i] != p.getCoefficients()[i]) {
317                 return false;
318             }
319         }
320         return true;
321     }

325      *
326      * @param x Point where we want to evaluate the value of the polynomial.
327      * @return The value of the polynomial at x.
328      */
329     public long evaluateAtX(long x) {
330         long value = 0;
331         for (int i = 0; i < this.coefficients.length; i++) {
332             value += this.coefficients[i] * (Math.pow(x, i));
333         }
334 
335         return value;

349      * Makes a copy of the present polynomial.
350      *
351      * @return A copy of this polynomial.
352      */
353     public Polynomial copyThisPoly() {
354         long[] initialArray = this.getCoefficients();
355         long[] reversedArray = new long[this.getCoefficients().length];
356         for (int i = 0; i < initialArray.length; i++) {
357             reversedArray[i] = initialArray[initialArray.length - i - 1];
358         }
359         Polynomial newPoly = new Polynomial(reversedArray);
360         return newPoly;
361     }

363     @Override
364     public String toString() {
365         StringBuilder s = new StringBuilder();
366         for (int i = this.coefficients.length - 1; i >= 0; i--) {
367             if (i == 0) {
368                 if (this.coefficients.length != 1) {
369                     s.append(" + ");
370                 }
371                 s.append(this.coefficients[i]);
372 
373             } else if (i == this.coefficients.length - 1) {
374                 s.append(this.coefficients[i]);
375                 s.append("x^");
376                 s.append(i);
377             } else {
378                 s.append(" + ");
379                 s.append(this.coefficients[i]);
380                 s.append("x^");
381                 s.append(i);
382             }
383         }
384         return s.toString();
385     }
