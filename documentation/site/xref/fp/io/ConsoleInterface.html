11      public ConsoleInterface(IO io) {

14      }
15  
16      public void run() throws IOException {
17          boolean isFirst = true;
18  
19          while (true) {
20              if (isFirst) {
21                  io.print("This is you number theoretic friend!");
22                  isFirst = false;
23              }
24  
25              System.out.println("Please select what you want to do.");
26              String command = io.readLine("Commands: [pc] (Primality check), [q] (quit)");
27  
28              if (command.equals("pc")) {
29                  io.print("Test whether an integer is a prime!");
30                  primalityTest(io);
31                  break;
32              }
33  
34              if (command.equals("quit")) {
35                  io.print("This was the best we could do.");
36                  break;
37              }
38          }
39      }
40  
41      private void primalityTest(IO io) {
42          String command = io.readLine("Give a number:");
43          long number = io.readLongInput(command);
44          if (pt.aksPrimalityTest(number)) {
45              io.print("Number " + number + " is indeed a prime!");
46          } else {
47              io.print("Nope.");
48          }
49      }
